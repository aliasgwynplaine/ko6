94/*
95 * fdt_strnlen: returns the length of a string or max_count - which ever is
96 * smallest.
97 * Input 1 string: the string whose size is to be determined
98 * Input 2 max_count: the maximum value returned by this function
99 * Output: length of the string or max_count (the smallest of the two)
100 */
101static inline size_t fdt_strnlen(const char *string, size_t max_count)
102{
103 const char *p = memchr(string, 0, max_count);
104 return p ? p - string : max_count;
105}
